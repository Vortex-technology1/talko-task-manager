rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // =============================================
    // STATISTICS RULES
    // =============================================
    // These rules should be MERGED with existing rules

    match /companies/{companyId}/metrics/{metricId} {
      // Anyone in company can read public/restricted metrics
      // owner_only metrics only readable by owner
      allow read: if isCompanyMember(companyId) && (
        resource.data.privacy != 'owner_only' || isOwner(companyId)
      );
      // Only owner/manager can create/update/delete metrics
      allow write: if isOwnerOrManager(companyId);
    }

    match /companies/{companyId}/metricEntries/{entryId} {
      // Read: scoped by role
      allow read: if isCompanyMember(companyId) && (
        isOwner(companyId) ||
        // Manager: can read all non-owner_only
        isManager(companyId) ||
        // Employee: only own entries
        resource.data.createdBy == request.auth.uid
      );
      // Write: member can create/update own entries, owner/manager can write any
      allow create: if isCompanyMember(companyId) && (
        request.resource.data.createdBy == request.auth.uid ||
        isOwnerOrManager(companyId)
      );
      allow update: if isCompanyMember(companyId) && (
        resource.data.createdBy == request.auth.uid ||
        isOwnerOrManager(companyId)
      );
      allow delete: if isOwnerOrManager(companyId);
    }

    match /companies/{companyId}/metricTargets/{targetId} {
      allow read: if isCompanyMember(companyId);
      allow write: if isOwnerOrManager(companyId);
    }

    match /companies/{companyId}/metricAggregates/{aggId} {
      // Aggregates: readable by all members, writable only by Cloud Functions (admin SDK)
      allow read: if isCompanyMember(companyId);
      allow write: if false; // Only Cloud Functions via admin SDK
    }

    match /companies/{companyId}/metricInsights/{insightId} {
      allow read: if isCompanyMember(companyId) && (
        isOwner(companyId) || isManager(companyId)
      );
      allow write: if false; // Only Cloud Functions
    }

    match /companies/{companyId}/metricAuditLog/{logId} {
      allow read: if isOwnerOrManager(companyId);
      allow write: if false; // Only Cloud Functions
    }

    // =============================================
    // HELPER FUNCTIONS (add to existing rules)
    // =============================================

    function isCompanyMember(companyId) {
      return request.auth != null &&
        exists(/databases/$(database)/documents/companies/$(companyId)/users/$(request.auth.uid));
    }

    function isOwner(companyId) {
      return request.auth != null &&
        get(/databases/$(database)/documents/companies/$(companyId)/users/$(request.auth.uid)).data.role == 'owner';
    }

    function isManager(companyId) {
      return request.auth != null &&
        get(/databases/$(database)/documents/companies/$(companyId)/users/$(request.auth.uid)).data.role == 'manager';
    }

    function isOwnerOrManager(companyId) {
      return isOwner(companyId) || isManager(companyId);
    }
  }
}
